#+end_src
* Stage 1: Installing Yay
#+begin_src bash
echo "###########################################################"
echo "###                   Installing Yay                    ###"
echo "###########################################################"
sudo pacman -S --noconfirm --needed git base-devel
git clone https://aur.archlinux.org/yay
cd yay
makepkg -si
cd ~
#+end_src
